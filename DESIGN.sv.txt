module counter_4bit (
    input clk,
    input rst,
    output reg [3:0] count
);
    // Trigger on the rising edge of clock or reset
    always @(posedge clk or posedge rst) begin
        if (rst)
            count <= 4'b0000; // Reset count to 0
        else
            count <= count + 1; // Increment count
    end
endmodule
